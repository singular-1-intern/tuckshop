﻿// TO GENERATE A MIGRATION
// =======================
// 'InitialMigration' is used as the name for the first migration
// For delta migrations, {Name} should be a short description of the feature/change

dotnet ef migrations add InitialMigration --project .\Tuckshop.AuthorisationServer\Tuckshop.AuthorisationServer.Models.Migrations --context AuthorisationDbContext

dotnet ef migrations add {Name} --project .\Tuckshop.AuthorisationServer\Tuckshop.AuthorisationServer.Models.Migrations --context AuthorisationDbContext

// TO ROLLBACK MIGRATIONS IN THE DB TO A SPECIFIC MIGRATION
// ========================================================
dotnet ef database update {PreviousMigrationName} --project .\Tuckshop.AuthorisationServer\Tuckshop.AuthorisationServer.Models.Migrations --context AuthorisationDbContext

// TO REMOVE THE LAST ADDED MIGRATION FROM CODE
// ============================================
dotnet ef migrations remove --project .\Tuckshop.AuthorisationServer\Tuckshop.AuthorisationServer.Models.Migrations --context AuthorisationDbContext
